لا تَسلْ عن سلامتِهْ

روحه فوق راحتِهْ

بدَّلَتْهُ همومُهُ

كفناً من وسادِتهْ

يَرقبُ الساعةَ التي

بعدَها هولُ ساعتِهْ

شاغلٌ فكرَ مَنْ يراهُ

بإطراقِ هامتِهْ

بيْنَ جنبيْهِ خافقٌ

يتلظَّى بغايتهْ

من رأى فَحْمةَ الدُّجى

أُضْرِمَتْ من شرارتِهْ

حَمَّلَتْهُ جهنَّمٌ

طَرفَاً من رسالتِه

هو بالباب واقفُ

والرًّدى منه خائفُ

فاهدأي يا عواصفُ

خجلاً من جراءتِهْ

صامتٌ لوْ تكلَّما

لَفَظَ النَّارَ والدِّما

قُلْ لمن عاب صمتَهُ

خُلِقَ الحزمُ أبكما

وأخو الحزم لم تزل

يدُهُ تسْبِقُ الفما

لا تلوموه قد رأى

منْهجَ الحقِّ مُظلما

وبلاداً أحبَّها

ركنُها قد تهدًّما

وخصوماً ببغْيِهمْ

ضجَّت الأَرضُ والسما

مرَّ حينٌ فكاد يق

تُلهُ اليأْسُ إنَّما

هو بالباب واقفُ

والرَّدى منه خائفُ

فاهدأي يا عواصفُ

خجلاً مِن جراءتِهْ

أطلت يا دهر نَحسي

متى تجوِد بسعدي

فقد تضاءل صبري

كما تعاظم وَجْدي

إذا تعشّقت هنداً

منحتني وصل دَعد

وأن تعشّقت دعداً

منحتني وصل هند

أما تعوّدت إلاّ

بأن تجود بضدّ

إني أريد عدوّي

فهات بعض أوُدّي

وجُد عليّ بوصل

فقد رضيت بصدّ

كلا فإن مقالي

هَزْل وليس بجِدّ

بل أنت أحقر عندي

من أن تجود وتُجدي

أني وأن كنت أشقى

بأوجُهٍ منك رُبْد

ربأت عنك بذمّي

كما ربأت بحمدي

إذ لست أنت بكُفْوِي

ولست أنت بنِدّي

لو كنت يا دهر حرّاً

وجئت تخدم عندي

لما أرتضيتك عبداً

ولا خويْدم عبد

وكيف أرضاك عبداً

وأنت أوغد وغد
